def self.from_yaml(name, yaml_data, opts = {})
    conf = YAML.load(yaml_data)
    configs = {}
    conf[name].each_pair do |ca_name, data|
        configs[ca_name] = R509::Config::CaConfig.load_from_hash(data, opts)
    end
    R509::Config::CaConfigPool.new(configs)
end
