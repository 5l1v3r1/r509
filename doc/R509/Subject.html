def initialize(arg=nil)
    case arg
    when Array
        @array = arg
    when OpenSSL::X509::Name
        sanitizer = R509::NameSanitizer.new
        @array = sanitizer.sanitize(arg)
    when R509::Subject
        @array = arg.to_a
    else
        @array = []
    end

    # see if X509 thinks this is okay
    name
end

def []=(key, value)
    added = false
    @array = @array.map{ |item|
        if key == item[0]
            added = true
            [key, value]
        else
            item
        end
    }

    if not added
        @array << [key, value]
    end

    # see if X509 thinks this is okay
    name

    @array
end
