def bit_length
  if self.rsa?
    return internal_obj.public_key.n.num_bits
  elsif self.dsa?
    return internal_obj.public_key.p.num_bits
  elsif self.ec?
    raise R509::R509Error, 'Bit length is not available for EC at this time.'
  end
end

def curve_name

def key_algorithm
  if self.rsa?
    "RSA"
  elsif self.dsa?
    "DSA"
  elsif self.ec?
    "EC"
  end
end
