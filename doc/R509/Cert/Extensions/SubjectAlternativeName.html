def initialize(*args)
  super(*args)

  @dns_names = self.value.scan( DNS_REGEX ).map { |match| match[0] }
  @ip_addresses = self.value.scan( IP_ADDRESS_REGEX ).map { |match| match[0] }
  @uris = self.value.scan( URI_REGEX ).map { |match| match[0] }
end
